# motion.div
framer-motion의 motion을 이용한 애니메이션 효과.
div는 element의 이름을 의미한다.
ul일 경우, motion.ul을 사용한다.

## Code
```jsx

import { AnimatePresence, motion } from 'framer-motion'

// element 자체를 없앴다가 보여주는 상황일 때
<AnimatePresence>
  {isOpen && (
    <motion.div
      className="popup-type-list__section-container"
      initial={{ translateX: '100%' }}
      animate={{ translateX: 0 }}
      exit={{ translateX: '100%' }}
      transition={{ type: 'easeOut', duration: 0.22 }}
    >
  )}
</AnimatePresence>

// element의 스타일만 변경하고자 할 때
// 아래의 내용은 slideUp slideDown 효과이다.
<motion.ul
  className={cn('lnb-filter-depth2__list overflow-hidden')}
  initial={{
    height: 0,
  }}
  animate={{
    height: parentItem.checked ? 'auto' : 0,
  }}
  transition={{ type: 'easeOut', duration: 0.15 }}
>
 {children}
</motion.ul>

```
